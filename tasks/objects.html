// Not just straight value for a property we can also have a raise
var person = {
  firstName: 'Matt',
  lastName: 'Miller',
  age: 26
  // This set is in an array
  children:['Mark','Amanda']
  // This set is another object
  address:{
    street:'555 Something st',
    city:'Boston',
    state: 'MA'
  },
  // Include functions
  // function called fullName
  fullName: function(){
    // The '.this' refers to is the current object that's in which is 'person' (var person = ..) above
    return this.firstName +" "+this.lastName;
  }
}
// to check this out open up the console in the inspect element
// since it's a function it has to have the parentheses ()
console.log(person.fullName());
